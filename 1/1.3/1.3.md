      <td>4</td>

    </tr>
    <tr>
      <td>5</td>
      <td>Comunicación de la decisión</td>

      <td>Supervisor de asistencia</td>
    </tr>
    <tr>
      <td>6</td>
      <td>Actualización del sistema</td>

      <td>3</td>

    </tr>

      <td>4</td>
      <td>Generación de reporte</td>

</table>
